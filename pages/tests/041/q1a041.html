void st(char s[]) {
    int i, j = 1;

    for (i = 0; i < 21; i++) {
        if (s[i] > ' '){
            s[i] -= j;
            if(j == 4)
                j = 1;
            else
                j++;
        }
    }
}
int ct( char s[] ) {
    int i = 0, j = 0, p = 1;
    char c;

    while (c = s[i++]) {
        if (c == ' ' && !p) {
            s[j++] = ' ';
            p = 1;
        } else if (c != ' ') {
            s[j++] = c;
            p = 0;
        }
    }
    if (p && j)
        --j;
    s[j] = '\0';

    return j;
}

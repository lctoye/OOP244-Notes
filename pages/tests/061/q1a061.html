What is the exact output of the following program? 

 #include <iostream>
 using namespace std;

 struct Vector {
     int x;
     int y;
     void display() const;
 };

 struct Set {
     Vector* v;
     int n;
 public:
     void ctor(int n, int* x, int* y);
     void display() const;
     Vector sum() const;
     void dtor();
 };

 void add(Vector& a, Vector* b) {
     a.x += b->x;
     a.y += b->y;
 }

 void Vector::display() const {
     cout << '{' << x << ',' << y << '}' << endl;
 }

 void Set::ctor(int k, int* x, int* y) {
     n = k;
     v = new Vector[n];
     for (int i = 0; i < n; i++) {
         v[i].x = x[i] + 1;
         v[i].y = y[i] - 1;
     }
 }

 void Set::display() const {
    for (int i = 0; i < n; i++)
        cout << '[' << v[i].x << ',' << v[i].y << "] ";
     cout << endl;
 }

 Vector Set::sum() const {
     Vector c = {0, 0};
     for (int i = 0; i < n; i++)
         add(c, &v[i]);
     return c;
 }

 void Set::dtor() {
     delete [] v;
     cout << n << endl;
 }

 int main() {
     int x[] = {0, 1, 0, 6};
     int y[] = {2, 1, 2, 2};
     Set s;
     s.ctor(3, y, x);
     s.display();
     s.sum().display();
     s.dtor();
 }

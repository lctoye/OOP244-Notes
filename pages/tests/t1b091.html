Question 2 - Walkthrough - (10 marks):

What is the exact output of the following program?  Show your rough work to avoid deductions. 

 #include <iostream>
 using namespace std;
 const int N = 2;

 class Fraction {
     int num;
     int den;
   public:
     Fraction();
     void set(int n, int d);
     void display() const;
     friend Fraction operator*(const Fraction& a, const Fraction& b); 
     ~Fraction();
 };
 Fraction::Fraction() {
    cout << '^'; num = 1; den = 1;
 }
 void Fraction::set(int n, int d) {
    den = (d != 0) ? d : 1; num = n;
 }
 void Fraction::display() const {
    cout << num << '/' << den << endl;
 }
 Fraction::~Fraction() {
    cout << '~'; display();
 }
 Fraction operator*(const Fraction& a, const Fraction& b) {
     Fraction f;
     f.set(a.num * b.num, a.den * b.den);
     return f;
 }

 int main() {
     Fraction f1, f2, f3;

     cout << "\n-----" << endl;
     f2.set(1, 2);
     f1.set(2, 3);
     cout << "-----" << endl;
     f1.display();
     f2.display();
     cout << "-----" << endl;
     f3 = f1 * f2;
     cout << "-----" << endl;
     cout << "=";
     f3.display();
     cout << "-----" << endl;

     return 0;
 }

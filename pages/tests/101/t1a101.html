Question 2 - Walkthrough (10 marks)

What is the exact output of the following program?  Show your rough work to avoid deductions. 

#include <iostream>
using namespace std;
const int MAX_CARDS = 3;

//--- Card.h -------------------------------------------------------------

class Card {
	int rank;
	char suit;
  public:
	Card();
	void set(char s, int r);
	void display() const;
	~Card();
};

//--- Card.cpp -----------------------------------------------------------

Card::Card() {
	rank = 0;
	suit = '?';
	cout << suit << rank;
}
void Card::set(char s, int r) {
	suit = s;
	rank = r;
}
void Card::display() const {
	cout << suit << rank;
}

Card::~Card() {
	cout << '~';
	display();
}


//--- Hand.h -------------------------------------------------------------

class Hand {
	int noCards;
	Card card[MAX_CARDS];
  public:
	Hand();
	void swap(int i, int j);
	void add(char s, int r);
	void display() const;
	~Hand();
};


//--- Hand.cpp -----------------------------------------------------------

Hand::Hand() {
	noCards = 0;
	cout << '$' << endl;
}

void Hand::swap(int i, int j) {
	if (i < noCards && j < noCards) {
		Card temp = card[i];
		card[i] = card[j];
		card[j] = temp;
	}
}

void Hand::add(char s, int r) {
	if (noCards < MAX_CARDS)
		card[noCards++].set(s, r);
}

void Hand::display() const {
	for (int i = 0; i < noCards; i++)
		card[i].display();
	cout << endl;
}

Hand::~Hand() {
	cout << '*';
}

//--- main.cpp -----------------------------------------------------------

int main() {
	Hand hand;

	hand.add('H', 9);
	hand.add('S', 6);
	hand.add('C', 3);
	hand.display();
	hand.swap(0,2);
	hand.swap(0,1);
	cout << endl;
	hand.display();

	return 0;
}

What is the exact output of the following program? 

 #include <iostream>
 #include <iomanip>
 #include <cstring>
 using namespace std;
 #define NEN 3
 #define NN  5
 #define NE  3
 #define NC  3

 struct Node {
 private:
     int label;
     int x;
     int y;
 public:
     void set(const char*);
     int getLabel() const {return label;}
     void display() const;
 };

 struct Element {
 private:
     char label[NC];
     int noOfNodes;
     int node[NEN];
 public:
     void set(const char*, int, int, int);
     int nNodes() const {return noOfNodes;}
     int getLabel(int) const;
     void display() const;
 };

 struct Mesh {
 private:
     int nNodes;
     Node node[NN];
     int nElements;
     Element element[NE];
     int index(int) const;
 public:
     void start() {nNodes = 0; nElements = 0;}
     void set(const char*);
     void set(const char*, int, int, int);
     void display() const;
 };

 int main() {
     Mesh mesh;

     mesh.start();
     mesh.set("1 0 0");
     mesh.set("2 1 0");
     mesh.set("3 1 1");
     mesh.set("4 0 1");
     mesh.set("5 1 2");
     mesh.set("1", 1, 2, 3);
     mesh.set("2", 1, 3, 4);
     mesh.set("3", 4, 3, 5);
     mesh.display();

     return 0;
 }

 void Node::set(const char* s) {
     bool empty;
     char c;
     int i, k, v;

     empty = true;
     i = 0;
     k = 0;
     v = 0;
     while ((c = s[i]) != '\0') {
         if (c == ' ')
             if (!empty) {
                 switch (k) {
                 case 0: label = v; break;
                 case 1: x = v; break;
                 case 2: y = v; break;
                 }
                 empty = true;
                 k++;
                 v = 0;
             }
         else {
             v = 10 * v + c - '0';
             empty = false;
         }
         i++;
     }
 }

 void Node::display() const {
     cout << setw(4) << x << setw(4) << y
          << ' ' << label << endl;
 }

 void Element::set(const char* s, int na, int nb, int nc) {
     strcpy(label, s);
     node[0] = na;
     node[1] = nb;
     node[2] = nc;
     noOfNodes = 3;
 }

 int Element::getLabel(int i) const {return node[i];}

 void Mesh::set(const char* s) {
     node[nNodes].set(s);
     nNodes++;
 }

 void Mesh::set(const char* s, int na, int nb, int nc) {
     element[nElements].set(s, na, nb, nc);
     nElements++;
 }

 int Mesh::index(int i) const {
     int found = 0;
     for (int k = 0; k < nNodes; k++)
         if (i == node[k].getLabel()) {
             found = k;
             k = NN;
         }
     return found;
 }

 void Mesh::display() const {
     for (int i = 0; i < nElements; i++) {
         for (int j = 0; j < element[i].nNodes(); j++)
             node[index(element[i].getLabel(j))].display();
         node[index(element[i].getLabel(0))].display();
         cout << endl;
     }
 }

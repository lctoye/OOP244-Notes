Question 3 - Walkthrough - (10 marks):

What is the exact output of the following program?  Show your rough work to avoid deductions. 

 #include <iostream>
 #include <iomanip>
 using namespace std;
 #define MK 3
 #define MT 4
 #define BR 60000
 #define SH 32

 class Key {
	int n;
	int m;
	bool on;
 public:
	Key() {n = 0; m = 0; on = false; cout << "K0 ";}
	Key(int k, int mm) {n = k; m = mm; on = false; cout << "K2 ";}
	~Key() {cout << '~' << n << ' ';}
	int  toggle() {on = !on; return on ? m : m + BR;}
	bool pressed() const {return on;}
	bool operator==(int k) const {return n == k;}
	friend ostream& operator<<(ostream& os, const Key& k) {
		char c = k.on ? '*' : ' ';
		os << c << setfill('0') << setw(2) << k.n << setfill(' ') << c; 
		return os;
	}
 };

 class KB {
	Key key[MK];
	int nKeys;
 public:
	KB(const Key* k, int n) {
		for (int i = 0; i < n; i++)
			key[i] = k[i];
		nKeys = n;
		cout << "KB ";
	}
	virtual ~KB() {cout << "~KB ";}
	virtual int toggle(int j) {
		int rc = 0;
		for (int i = 0; i < nKeys; i++)
			if (key[i] == j)
				rc = key[i].toggle();
		return rc;
	}
	bool pressed(int j) const {
		bool rc = false;
		for (int i = 0; i < nKeys; i++)
			if (key[i] == j)
				rc = key[i].pressed();
		return rc;
	}
	friend ostream& operator<<(ostream& os, const KB& p) {
		for (int i = 0; i < p.nKeys; i++)
			os << p.key[i] << ' ';
		return os;
	}
 };

 class KBWS : public KB {
	bool down;
	int shift;
 public:
	KBWS(const Key* k, int n, int s) : KB(k, n) {
		down = false; shift = s; cout << "KBWS ";
	}
	~KBWS() {cout << "~KBWS ";}
	int toggle(int j) {
		int rc;
		if (shift == j) {
			down = !down;
			rc = KB::toggle(j);
		} else if (pressed(j))
			rc = KB::toggle(j);
		else
			rc = KB::toggle(j) - (down ? SH: 0);
		return rc;
	}
 };

 void process(KB& v) {
	int k[] = {22, 44, 22, 50};
	for (int i = 0; i < MT; i++) {
		cout << setw(5) << v.toggle(k[i]) << ": ";
		cout << v << endl;
	}
 }

 int main() {
	Key y[MK];
	y[0] = Key(44,34);
	y[1] = Key(50,49);
	y[2] = Key(22,53);
	cout << endl;
	KBWS v(y, MK, 44);
	cout << endl << "    0: " << v << endl;
	process(v);
	return 0;
 }

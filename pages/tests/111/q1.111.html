#include 
using namespace std;
const int MAX_CHARS = 4;
const int MAX_APPS = 2;

struct App {
  private:
    char str[MAX_CHARS + 1];
  public:
    void set(const char* s);
    void play();
    void display() const;
};
struct Phone {
  private:
    int noApps;
    App app[MAX_APPS];
  public:
    void start();
    void add(const char* s);
    void play();
    void display() const;
};

int main() {
    Phone iPhone;
    iPhone.start();
    iPhone.add("Vkog");
    iPhone.display();
    iPhone.play();
    iPhone.add("Dbmd");
    iPhone.display();
    iPhone.play();
    iPhone.add("Ring");
    iPhone.display();
};

void App::set(const char* s) {
    int i;
    for (i = 0; s[i] != '\0' && i < MAX_CHARS; i++)
        str[i] = s[i];
    str[i] = '\0';
}
void App::play() {
    for (int i = 0; str[i] != '\0'; i++)
        str[i] = str[i] - 1;
}
void App::display() const {
    cout << str << endl;
}

void Phone::start() { noApps = 0; }
void Phone::add(const char* s) {
    if (noApps < MAX_APPS)
        app[noApps++].set(s);
}
void Phone::play() {
    for (int i = 0; i < noApps; i++)
        app[i].play();
}
void Phone::display() const {
    for (int i = 0; i < noApps; i++)
        app[i].display();
}

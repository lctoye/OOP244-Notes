What is the exact output of the following program? Include rough work to avoid deductions. 

 #include <iostream>
 using namespace std;
 const int MP = 2;
 const int MN = 4;

 class Passenger {
     char name[MN + 1];
   public:
     Passenger();
     Passenger(const char*);
     Passenger(const Passenger&);
     Passenger& operator=(const Passenger& s);
     ~Passenger();
     friend ostream& operator<<(ostream& os, const Passenger& p);
 };
 Passenger::Passenger() {
     cout << 'P';
     name[0] = '\0';
 }
 Passenger::Passenger(const char* n) {
     cout << 'Q';
     strncpy(name, n, MN);
     name[MN] = '\0';
 }
 Passenger::Passenger(const Passenger& src) {
     cout << '*';
     *this = src;
 }
 Passenger& Passenger::operator=(const Passenger& src) {
     cout << '=';
     strcpy(name, src.name);
     return *this;
 }
 Passenger::~Passenger() {
     cout << '~' << *this << endl;
 }
 ostream& operator<<(ostream& os, const Passenger& p) {
     return os << ':' << p.name;
 }

 class Car {
     int np;
     Passenger pass[MP];
   public:
     Car();
     void operator+=(const Passenger p);
     friend ostream& operator<<(ostream& os, const Car& s);
     ~Car();
 };
 Car::Car() {
     cout << 'C' << endl;
     np = 0;
 }
 void Car::operator+=(const Passenger p) {
     if (np < MP)
         pass[np++] = p;
 }
 Car::~Car() {
     cout << '#' << *this;
 }
 ostream& operator<<(ostream& os, const Car& c) {
     for (int i = 0; i < c.np; i++)
         os << c.pass[i];
     os << endl;
     return os;
 }

 int main() {
     Car car;
     Passenger p1("John"), p2("Dori");
     cout << endl;
     car += p2;
     cout << car;
     car += p1;
     cout << car;
 }

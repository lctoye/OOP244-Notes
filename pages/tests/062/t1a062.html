Question 1 - Word Problem - (10 marks):

Design and code a class named Flight that holds information about an airline flight.  Upon instantiation, a Flight object may receive an int holding the flight number, and a null-terminated C-style string holding the flight destination.  The flight number should be positive-valued.  If the object receives a valid number and a string, the object accepts both.  Otherwise, the object assumes a safe empty state.  The object adopts a default output width of ten (10) characters for the flight number.  Your class does not impose any limitation on the number of characters in a Flight object.  Your design includes all of the member functions needed to accomodate this feature under a variety of applications, including the one below.

Your class also includes the following functions:

  • const char* destination() const - a query that returns the address of the first non-whitepsace character in the description of the Flight object;
  • int number() const - a query that returns the flight number; and
  • Flight& width(int) - a modifier that changes the width of the output field for the flight number to the integer value received if that value is a greater than that needed to display fully the flight number; if the integer value received is invalid, your object reverts to the default number of characters for the class.  Your function returns a reference to the current object as modified, so that your function can be used to insert the current object into the output stream directly as shown below; and
  • an insertion operator that inserts the flight number followed by the flight description into an output stream.  Your operator displays the flight number left-justified in the current field width, followed by a trailing space and the flight description.  Your operator accepts cascading. 
For example, consider the following program that uses your class 

 #include <iostream>
 using namespace std;
 #include "Flight.h"

 void display(Flight flight) {

     cout << flight.width(2) << endl;

 }

 int main ( ) {
     Flight f, g(857, "Toronto"); 

     cout << g.width(9) << endl;
     f = g;
     display(f);
     cout << f.number() << endl;
     cout << f.description() << endl;
     return 0;
 }

This program produces the following output 

 857       Toronto
 857        Toronto
 857
 Toronto


Question 2 - Walkthrough - (10 marks):

What is the exact output of the following program?  Show your rough work to avoid deductions. 

 #include <iostream>
 using namespace std;

 #define GST 0.07
 #define PST 0.08

 class Total; // tells compiler Total will be defined below 

 class SaleLineItem {
     int quantity;
     double unitPrice;
     char taxable;
 public:
     SaleLineItem();
     SaleLineItem(int, double, char);
     ~SaleLineItem();
     void operator++();
     void operator++(int);
     Total display() const;
 };

 class Total {
     double total;
     double gst;
     double pst;
     double subtotal;
 public:
     Total();
     Total(double, char);
     Total(const Total& t);
     void operator+=(Total);
     void display() const;
     ~Total();
 };

 class Sale {
     SaleLineItem item[4];
     int noItems;
 public:
     Sale();
     void display() const;
     void operator+=(const SaleLineItem& item);
     ~Sale();
 };

 int main ( ) {
     SaleLineItem item[3];
     item[0] = SaleLineItem(3, 1.50, '-');
     item[1] = SaleLineItem(2, 0.50, 'b');
     item[2] = SaleLineItem(1, 2.00, 'g');

     Sale sale;
     for (int i = 0; i < 3; i++)
         sale += item[i];
     sale.display();
     item[1]++;

     return 0;
 }

 Sale::Sale() {
     noItems = 0;
     cout << " S: " << endl;
 }

 void Sale::display() const {
     Total total;

     cout << "Sale Receipt" << endl;
     for (int i = 0; i < noItems; i++)
         total += item[i].display();
     total.display();
     cout << "------------" << endl;
 }

 void Sale::operator+=(const SaleLineItem& lineItem) {
     item[noItems++] = lineItem;
 }

 Sale::~Sale() {
     cout << "~S: " << noItems << endl;
 }

 SaleLineItem::SaleLineItem() {
     quantity = 0;
     unitPrice = 0.0;
     taxable = ' ';
 }

 SaleLineItem::SaleLineItem(int q, double u, char t) {
     quantity = q;
     unitPrice = u;
     taxable = t;
 }

 void SaleLineItem::operator++() {
     quantity += 2;
 }

 void SaleLineItem::operator++(int) {
     quantity++;
 }

 SaleLineItem::~SaleLineItem() {
     cout << "~L: " << quantity << endl;
 }

 Total SaleLineItem::display() const {
     double price = quantity * unitPrice;

     cout << "SR: ";
     cout << quantity << '@' << unitPrice << " = "
      << price<< ' ' << taxable << endl;
     Total total(price, taxable);
     return total;
 }

 Total::Total() {
     subtotal = 0;
     gst      = 0;
     pst      = 0;
     total    = 0;
     cout << " T: " << endl;
 }

 Total::~Total() {
     cout << "~T: " << total << endl;
 }

 Total::Total(const Total& t) {
     subtotal = t.subtotal;
     gst      = t.gst;
     pst      = t.pst;
     total    = t.total;
     cout << "CT: " << total << endl;
 }

 Total::Total(double s, char t) {
     subtotal = s;
     if (t == 'g' || t == 'b')
         gst      = GST * s;
     else
         gst = 0;
     if (t == 'p' || t == 'b')
         pst      = PST * s;
     else
         pst = 0;
     total    = s + gst + pst;
     cout << " T: " << total << endl;
 }

 void Total::display() const {

     cout << "SR: " << subtotal << endl;
     cout << "SR: " << gst << endl;
     cout << "SR: " << pst << endl;
     cout << "SR: " << total << endl;
 }

 void Total::operator+=(Total t) {
     subtotal += t.subtotal;
     gst      += t.gst;
     pst      += t.pst;
     total    += t.total;
     cout << "+T: " << total << endl;
 }

What is the exact output of the following program? 

 #include <iostream>
 #include <iomanip>
 using namespace std;
 const int N = 2;
 const double TAX = 0.10;
 class Price {
     int no;
     double price;
   public:
     Price();
     Price(int n, double p);
     Price(const Price& p);
     Price& operator=(const Price& p);
     void operator+=(const Price& p);
     void operator*=(double x);
     friend ostream& operator<<(ostream& os, const Price& p);
     ~Price();
 };
 Price::Price() {
     no = 0;
     price = 0.0;
     cout << '%';
 }
 Price::Price(int n, double p) {
     no = n;
     price = p;
     cout << '#';
 }
 Price::Price(const Price& p) {
     no = p.no;
     price = p.price;
     cout << '^';
 }
 Price& Price::operator=(const Price& p) {
     no = p.no;
     price = p.price;
     cout << '=';
     return *this;
 }
 void Price::operator+=(const Price& p) {
     price += p.price;
     cout << '+';
 }
 void Price::operator*=(double x) {
     price *= x;
     cout << '*';
 }
 Price::~Price() {
     cout << '~' << *this << endl;
 }
 ostream& operator<<(ostream& os, const Price& p) {
     os << setw(2) << setfill('0') << p.no << setfill('.') << fixed
        << setprecision(2) << setw(5) << p.price;
     return os;
 }
 Price getTax(const Price& p) {
     Price t = p;
     t *= TAX;
     cout << endl;
     return t;
 }
 int main() {
     Price a[N], total;
     cout << "\n----------\n"; // 10 dashes
     a[0] = Price(1, 3.50);
     a[1] = Price(2, 2.30);
     cout << '-';
     for (int i = 0; i < N; i++) {
         cout << a[i] << endl;
         total += a[i];
     }
     cout << total << endl;
     cout << "---------\n";  // 9 dashes
     total += getTax(total);
     cout << "::" << total << endl;
     cout << "---------\n";  // 9 dashes
     return 0;
 }

Animal Module

Animal.h

 // Virtual Functions
 // Animal.h

 #include <iostream>

 class Animal {
   public:
     virtual void display() const;
 };

 class Horse : public Animal {
   public:
     void display() const;
 };

Animal.cpp

 // Virtual Functions
 // Animal.cpp

 #include "Animal.h"

 void Animal::display() const {
     std::cout << "Animal" << std::endl; 
 }

 void Horse::display() const {
     Animal::display();
     std::cout << "Horse" << std::endl;
 }
